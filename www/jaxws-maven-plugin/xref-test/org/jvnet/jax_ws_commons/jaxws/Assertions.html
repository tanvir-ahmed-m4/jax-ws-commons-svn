20  import java.io.BufferedReader;
21  import java.io.File;
22  import java.io.FileReader;
23  import java.io.IOException;
24  import java.util.zip.ZipException;
25  import java.util.zip.ZipFile;
26  import org.testng.Assert;
27  
28  /**
29   *
30   * @author lukas

32  public class Assertions {
33  
34      public static void assertFilePresent(File project, String path) {
35          File f = new File(project, path);
36          Assert.assertTrue(f.exists(), "Not found " + f.getAbsolutePath());
37      }
38  
39      public static void assertFileNotPresent(File project, String path) {
40          File f = new File(project, path);
41          Assert.assertFalse(f.exists(), "Found " + f.getAbsolutePath());
42      }
43  
44      public static void assertFileContains(File project, String path, String s) throws IOException {
45          File f = new File(project, path);
46          Assert.assertTrue(f.exists());
47          Assert.assertTrue(f.isFile());
48          BufferedReader r = new BufferedReader(new FileReader(f));
49          String line;
50          while ((line = r.readLine()) != null) {
51              if (line.contains(s)) {
52                  return;
53              }
54          }
55          Assert.fail("'" + s + "' is missing in:" + f.getAbsolutePath());
56      }
57  
58      public static void assertJarContains(File project, String jarName, String path) throws ZipException, IOException {
59          File f = new File(project, "target/" + jarName);
60          Assert.assertTrue(f.exists());
61          Assert.assertTrue(f.isFile());
62          ZipFile zf = new ZipFile(f);
63          Assert.assertNotNull(zf.getEntry(path), "'" + path + "' is missing in: " + jarName);
64      }
65  
66      public static void assertJarNotContains(File project, String jarName, String path) throws ZipException, IOException {
67          File f = new File(project, "target/" + jarName);
68          Assert.assertTrue(f.exists());
69          Assert.assertTrue(f.isFile());
70          ZipFile zf = new ZipFile(f);
71          Assert.assertNull(zf.getEntry(path), "'" + path + "' is in: " + jarName);
72      }
73  }
